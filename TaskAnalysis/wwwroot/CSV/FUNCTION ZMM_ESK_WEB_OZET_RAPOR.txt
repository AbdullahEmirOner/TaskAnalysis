FUNCTION ZMM_ESK_WEB_OZET_RAPOR.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     VALUE(I_YIL) TYPE  NUMC4
*"     VALUE(I_DONEM_1) TYPE  NUMC2
*"     VALUE(I_DONEM_2) TYPE  NUMC2 OPTIONAL
*"     VALUE(I_MENSE) TYPE  ZMM_ESK_MENSE
*"     VALUE(I_OY_SON_DONEM) TYPE  CHAR1 OPTIONAL
*"     VALUE(I_OY_SON_MG) TYPE  CHAR1 OPTIONAL
*"     VALUE(I_BIRIKIMLI) TYPE  CHAR1 OPTIONAL
*"     VALUE(I_BIRIKIMSIZ) TYPE  CHAR1 OPTIONAL
*"     VALUE(I_BIRIKIMSIZ_KUMULATIF) TYPE  CHAR1 OPTIONAL
*"     VALUE(I_ASKERI_TICARI) TYPE  ZMM_ESK_ASKERI_TICARI OPTIONAL
*"     VALUE(I_ESK_DONEM_1) TYPE  SPMON OPTIONAL
*"     VALUE(I_ESK_DONEM_2) TYPE  SPMON OPTIONAL
*"  EXPORTING
*"     VALUE(E_ERROR) TYPE  CHAR1
*"     VALUE(E_MESSAGE) TYPE  CHAR200
*"  TABLES
*"      ET_SONUC STRUCTURE  ZMM_ESK_S_WEB_OZET
*"      ET_SATICI_SONUC STRUCTURE  ZMM_ESK_S_WEB_SATICI
*"      IT_KATEGORI STRUCTURE  ZMM_ESK_S_WEB_KATEGORI OPTIONAL
*"----------------------------------------------------------------------
  TYPES: TY_DEC23_5 TYPE ZMM_ESK_CIRO_BAZ_FIYAT.

  TYPES: BEGIN OF TY_SONUC_ORAN,
          DONEM           TYPE ZMM_ESK_DONEM,
          LIFNR           TYPE LIFNR,
          NAME1           TYPE NAME1_LF,
          BUYER           TYPE ZMM_ESK_BUYER,
          BUYER_ADI       TYPE AD_NAMTEXT,
          MENSE           TYPE ZMM_ESK_MENSE,
          KATEGORI        TYPE ZMM_ESK_KTGR_KOD,
          UYGULANAN       TYPE ZMM_ESK_ORAN_2,
          ESKALASYON      TYPE ZMM_ESK_ORAN_2,
          VRM_ESKALASYON  TYPE ZMM_ESK_ORAN_2,
          HEDEF           TYPE ZMM_ESK_ORAN_2,
          ANLASMA         TYPE ZMM_ESK_ORAN_2,
          PERFORMANS      TYPE ZMM_ESK_ORAN_2,
          ESK_PERFORMANS  TYPE ZMM_ESK_ORAN_2,
          CIRO            TYPE WERTV13,
          CIRO_PAYI       TYPE ZMM_ESK_ORAN_2,
         END OF TY_SONUC_ORAN,
         BEGIN OF TY_TOP_ORAN,
          DONEM           TYPE ZMM_ESK_DONEM,
          MENSE           TYPE ZMM_ESK_MENSE,
          KATEGORI        TYPE ZMM_ESK_KTGR_KOD,
          UYGULANAN       TYPE ZMM_ESK_ORAN_2,
          ESKALASYON      TYPE ZMM_ESK_ORAN_2,
          VRM_ESKALASYON  TYPE ZMM_ESK_ORAN_2,
          HEDEF           TYPE ZMM_ESK_ORAN_2,
          ANLASMA         TYPE ZMM_ESK_ORAN_2,
          PERFORMANS      TYPE ZMM_ESK_ORAN_2,
          ESK_PERFORMANS  TYPE ZMM_ESK_ORAN_2,
         END OF TY_TOP_ORAN,
         BEGIN OF TY_ESK_SONUC.
          INCLUDE STRUCTURE ZMM_ESK_SONUC.
  TYPES:  TTR_BAZ_BIRIM_FIYAT  TYPE TY_DEC23_5, "WERTV13,
          TTR_UYGULANAN_FIYAT  TYPE TY_DEC23_5, "WERTV13,
          TTR_ESKALASYON_FIYAT TYPE TY_DEC23_5, "WERTV13,
          TTR_VRM_ESK_FIYATI   TYPE TY_DEC23_5, "WERTV13,
          TTR_HEDEF_FIYATI     TYPE TY_DEC23_5, "WERTV13,
          TTR_ANLASMA_BRM_FYT  TYPE TY_DEC23_5, "WERTV13,
         END OF TY_ESK_SONUC,
         BEGIN OF TY_TOP_SATICI,
          DONEM            TYPE ZMM_ESK_DONEM,
          MENSE            TYPE ZMM_ESK_MENSE,
          LIFNR            TYPE LIFNR,
          BUYER            TYPE ZMM_ESK_BUYER,
          KATEGORI         TYPE ZMM_ESK_KTGR_KOD,
          BAZ_BIRIM_FIYAT  TYPE TY_DEC23_5, "WERTV13,
          UYGULANAN_FIYAT  TYPE TY_DEC23_5, "WERTV13,
          ESKALASYON_FIYAT TYPE TY_DEC23_5, "WERTV13,
          VRM_ESK_FIYATI   TYPE TY_DEC23_5, "WERTV13,
          HEDEF_FIYATI     TYPE TY_DEC23_5, "WERTV13,
          ANLASMA_BRM_FYT  TYPE TY_DEC23_5, "WERTV13,
         END OF TY_TOP_SATICI,
         BEGIN OF TY_TOP_TUTAR,
          DONEM            TYPE ZMM_ESK_DONEM,
          MENSE            TYPE ZMM_ESK_MENSE,
          KATEGORI         TYPE ZMM_ESK_KTGR_KOD,
          BAZ_BIRIM_FIYAT  TYPE TY_DEC23_5, "WERTV13,
          UYGULANAN_FIYAT  TYPE TY_DEC23_5, "WERTV13,
          ESKALASYON_FIYAT TYPE TY_DEC23_5, "WERTV13,
          VRM_ESK_FIYATI   TYPE TY_DEC23_5, "WERTV13,
          HEDEF_FIYATI     TYPE TY_DEC23_5, "WERTV13,
          ANLASMA_BRM_FYT  TYPE TY_DEC23_5, "WERTV13,
         END OF TY_TOP_TUTAR,
         BEGIN OF TY_LIFNR,
           LIFNR TYPE LIFNR,
           NAME1 TYPE NAME1_LF,
         END OF TY_LIFNR,
         BEGIN OF TY_BUYER,
           BNAME TYPE XUBNAME,
           NAME_TEXT TYPE AD_NAMTEXT,
         END OF TY_BUYER.

  DATA: LV_HESAPLAMA_TIPI TYPE ZMM_ESK_HESAPLAMA_TIPI,
        LV_BAS_DONEM      TYPE ZMM_ESK_DONEM,
        LV_BITIS_DONEM    TYPE ZMM_ESK_DONEM,
        LT_ESK_SONUC      TYPE TABLE OF TY_ESK_SONUC,
        LS_ESK_SONUC      TYPE TY_ESK_SONUC,
        LV_LIFNR          TYPE LIFNR,
        LV_MENSE          TYPE ZMM_ESK_MENSE,
        LT_TOP_SATICI     TYPE TABLE OF TY_TOP_SATICI,
        LS_TOP_SATICI     TYPE TY_TOP_SATICI,
        LT_SONUC_ORAN     TYPE TABLE OF TY_SONUC_ORAN,
        LS_SONUC_ORAN     TYPE TY_SONUC_ORAN,
        LT_TOP_TUTAR      TYPE TABLE OF TY_TOP_TUTAR,
        LS_TOP_TUTAR      TYPE TY_TOP_TUTAR,
        LT_TOP_ORAN       TYPE TABLE OF TY_TOP_ORAN,
        LS_TOP_ORAN       TYPE TY_TOP_ORAN,
        LT_LIFNR          TYPE SORTED TABLE OF TY_LIFNR
                          WITH NON-UNIQUE KEY LIFNR,
        LS_LIFNR          TYPE TY_LIFNR,
        LT_BUYER          TYPE TABLE OF TY_BUYER,
        LS_BUYER          TYPE TY_BUYER,
        LV_TOPLAM_TUTAR   TYPE WERTV13.
  DATA: R_ASKERI_TICARI   TYPE RANGE OF ZMM_ESK_ASKERI_TICARI
                          WITH HEADER LINE.
  DATA: R_KATEGORI        TYPE RANGE OF ZMM_ESK_KTGR_KOD
                          WITH HEADER LINE.

  CONSTANTS: C_TUMU  TYPE ZMM_ESK_MENSE VALUE '0', " Tümü
             C_ITHAL TYPE ZMM_ESK_MENSE VALUE '1', " İthal
             C_YERLI TYPE ZMM_ESK_MENSE VALUE '2', " Yerli
             C_OYSD_BIRIKIMSIZ  TYPE ZMM_ESK_HESAPLAMA_TIPI VALUE '1', " Önc. yıl son dönem - birikimsiz
             C_OYSD_BIRIKIMLI   TYPE ZMM_ESK_HESAPLAMA_TIPI VALUE '2', " Önc. yıl son dönem - birikimli
             C_OYSMG_BIRIKIMSIZ TYPE ZMM_ESK_HESAPLAMA_TIPI VALUE '3', " Önc. yıl son mal giriş tarihi - birikimsiz
             C_OYSMG_BIRIKIMLI  TYPE ZMM_ESK_HESAPLAMA_TIPI VALUE '4'. " Önc. yıl son mal giriş tarihi - birikimli


  IF I_YIL IS INITIAL.

    E_ERROR = 'X'.
    E_MESSAGE = 'Yıl alanı boş geçilemez!'.
    RETURN.

  ENDIF.
  IF I_DONEM_1 IS INITIAL.

    E_ERROR = 'X'.
    E_ERROR = 'Başlangıç dönemi alanı boş geçilemez!'.
    RETURN.

  ENDIF.
  IF I_OY_SON_DONEM IS INITIAL AND
     I_OY_SON_MG IS INITIAL.

    E_ERROR   = 'X'.
    E_MESSAGE = 'Baz fiyat belirleme yöntemi seçilmelidir!'.
    RETURN.

  ELSEIF I_OY_SON_DONEM IS NOT INITIAL AND
         I_OY_SON_MG IS NOT INITIAL.

    E_ERROR   = 'X'.
    E_MESSAGE = 'Yalnız bir tane baz fiyat belirleme yöntemi seçilmelidir!'.
    RETURN.

  ENDIF.
  IF I_BIRIKIMLI IS INITIAL AND
     I_BIRIKIMSIZ IS INITIAL AND
     I_BIRIKIMSIZ_KUMULATIF IS INITIAL.

    E_ERROR   = 'X'.
    E_MESSAGE = 'Mal giriş miktarı belirleme yöntemi seçimelidir!'.
    RETURN.

  ELSEIF I_BIRIKIMLI IS NOT INITIAL AND
         I_BIRIKIMSIZ IS NOT INITIAL.

    E_ERROR   = 'X'.
    E_MESSAGE = 'Yalnız bir tane mal giriş miktarı belirleme yöntemi seçilmelidir!'.
    RETURN.
  ELSEIF ( I_BIRIKIMLI IS NOT INITIAL OR
           I_BIRIKIMSIZ IS NOT INITIAL ) AND I_BIRIKIMSIZ_KUMULATIF IS NOT INITIAL.

    E_ERROR   = 'X'.
    E_MESSAGE = 'Yalnız bir tane mal giriş miktarı belirleme yöntemi seçilmelidir!'.
    RETURN.
  ENDIF.


  " Dönem aralığının belirlenmesi
  CONCATENATE I_YIL I_DONEM_1 INTO LV_BAS_DONEM.

  IF I_DONEM_2 IS INITIAL.
    LV_BITIS_DONEM = LV_BAS_DONEM.
  ELSE.
    CONCATENATE I_YIL I_DONEM_2 INTO LV_BITIS_DONEM.
  ENDIF.

*<< Farklı yıllardan veri çekilmek isteniyorsa kullanılır
  IF I_ESK_DONEM_1 IS NOT INITIAL AND I_ESK_DONEM_2 IS NOT INITIAL.
    LV_BAS_DONEM = I_ESK_DONEM_1.
    LV_BITIS_DONEM = I_ESK_DONEM_2.
  ENDIF.

  " Hesaplama tipinin belirlenmesi

  IF I_OY_SON_DONEM EQ 'X'.

    IF I_BIRIKIMSIZ EQ 'X'.

      LV_HESAPLAMA_TIPI = C_OYSD_BIRIKIMSIZ.

    ELSEIF I_BIRIKIMLI EQ 'X'.

      LV_HESAPLAMA_TIPI = C_OYSD_BIRIKIMLI.

    ENDIF.

  ELSEIF I_OY_SON_MG EQ 'X'.

    IF I_BIRIKIMSIZ EQ 'X'.

      LV_HESAPLAMA_TIPI = C_OYSMG_BIRIKIMSIZ.

    ELSEIF I_BIRIKIMLI EQ 'X'.

      LV_HESAPLAMA_TIPI = C_OYSMG_BIRIKIMLI.

    ENDIF.

  ENDIF.

  IF I_BIRIKIMSIZ_KUMULATIF EQ 'X'.
    IF I_OY_SON_DONEM EQ 'X'.
      LV_HESAPLAMA_TIPI = C_OYSD_BIRIKIMSIZ.
    ELSE.
      E_ERROR   = 'X'.
      E_MESSAGE = 'Birikimsiz Kümülatif için "SON DÖNEM" seçilmelidir!'.
      RETURN.
    ENDIF.
  ENDIF.

  LOOP AT IT_KATEGORI.
    R_KATEGORI-SIGN   = 'I'.
    R_KATEGORI-OPTION = 'EQ'.
    R_KATEGORI-LOW    = IT_KATEGORI-KATEGORI.
    APPEND R_KATEGORI.
  ENDLOOP.

  IF I_ASKERI_TICARI IS NOT INITIAL.
    R_ASKERI_TICARI-SIGN   = 'I'.
    R_ASKERI_TICARI-OPTION = 'EQ'.
    R_ASKERI_TICARI-LOW    = I_ASKERI_TICARI.
    APPEND R_ASKERI_TICARI.
  ENDIF.

  IF I_MENSE EQ C_ITHAL OR
     I_MENSE EQ C_YERLI .

    SELECT * FROM ZMM_ESK_SONUC
      INTO CORRESPONDING FIELDS OF TABLE LT_ESK_SONUC
      WHERE DONEM BETWEEN LV_BAS_DONEM AND LV_BITIS_DONEM
        AND MENSE EQ I_MENSE
        AND HESAPLAMA_TIPI EQ LV_HESAPLAMA_TIPI
        AND KATEGORI       IN R_KATEGORI
        AND ASKERI_TICARI  IN R_ASKERI_TICARI.

  ELSE.

    SELECT * FROM ZMM_ESK_SONUC
      INTO CORRESPONDING FIELDS OF TABLE LT_ESK_SONUC
      WHERE DONEM BETWEEN LV_BAS_DONEM AND LV_BITIS_DONEM
        AND HESAPLAMA_TIPI EQ LV_HESAPLAMA_TIPI
        AND KATEGORI       IN R_KATEGORI
        AND ASKERI_TICARI  IN R_ASKERI_TICARI.

  ENDIF.

  IF LT_ESK_SONUC[] IS INITIAL.

    E_MESSAGE = 'Girilen kriterler için veritabanında kayıt bulunamadı.'.
    RETURN.

  ENDIF.

  SELECT LIFNR NAME1 FROM LFA1
    INTO TABLE LT_LIFNR
    FOR ALL ENTRIES IN LT_ESK_SONUC
    WHERE LIFNR EQ LT_ESK_SONUC-LIFNR.

  SELECT BNAME NAME_TEXT FROM V_USR_NAME
    INTO TABLE LT_BUYER
    FOR ALL ENTRIES IN LT_ESK_SONUC
    WHERE BNAME EQ LT_ESK_SONUC-BUYER.



  CLEAR:LV_LIFNR,
        LV_MENSE,
        LV_TOPLAM_TUTAR.

  SORT LT_ESK_SONUC BY LIFNR DONEM.
  DELETE LT_ESK_SONUC WHERE BAZ_BIRIM_FIYAT IS INITIAL.

  LOOP AT LT_ESK_SONUC INTO LS_ESK_SONUC.

    IF LS_ESK_SONUC-LIFNR NE LV_LIFNR.
      LV_LIFNR = LS_ESK_SONUC-LIFNR.
      LV_MENSE = LS_ESK_SONUC-MENSE.
    ENDIF.

    LS_ESK_SONUC-TTR_BAZ_BIRIM_FIYAT  = LS_ESK_SONUC-BAZ_BIRIM_FIYAT  * LS_ESK_SONUC-MAL_GIRIS_MIKTAR.
    LS_ESK_SONUC-TTR_UYGULANAN_FIYAT  = LS_ESK_SONUC-UYGULANAN_FIYAT  * LS_ESK_SONUC-MAL_GIRIS_MIKTAR.
    LS_ESK_SONUC-TTR_ESKALASYON_FIYAT = LS_ESK_SONUC-ESKALASYON_FIYAT * LS_ESK_SONUC-MAL_GIRIS_MIKTAR.
    LS_ESK_SONUC-TTR_VRM_ESK_FIYATI   = LS_ESK_SONUC-VRM_ESK_FIYATI   * LS_ESK_SONUC-MAL_GIRIS_MIKTAR.
    LS_ESK_SONUC-TTR_HEDEF_FIYATI     = LS_ESK_SONUC-HEDEF_FIYATI     * LS_ESK_SONUC-MAL_GIRIS_MIKTAR.
    LS_ESK_SONUC-TTR_ANLASMA_BRM_FYT  = LS_ESK_SONUC-ANLASMA_BRM_FYT  * LS_ESK_SONUC-MAL_GIRIS_MIKTAR.

    MODIFY LT_ESK_SONUC FROM LS_ESK_SONUC.

    CLEAR:LS_TOP_SATICI.

    LS_TOP_SATICI-DONEM             = LS_ESK_SONUC-DONEM.
    LS_TOP_SATICI-MENSE             = LV_MENSE.
    LS_TOP_SATICI-LIFNR             = LS_ESK_SONUC-LIFNR.
    LS_TOP_SATICI-BUYER             = LS_ESK_SONUC-BUYER.
    LS_TOP_SATICI-KATEGORI          = LS_ESK_SONUC-KATEGORI.
    LS_TOP_SATICI-BAZ_BIRIM_FIYAT   = LS_ESK_SONUC-TTR_BAZ_BIRIM_FIYAT .
    LS_TOP_SATICI-UYGULANAN_FIYAT   = LS_ESK_SONUC-TTR_UYGULANAN_FIYAT .
    LS_TOP_SATICI-ESKALASYON_FIYAT  = LS_ESK_SONUC-TTR_ESKALASYON_FIYAT.
    LS_TOP_SATICI-VRM_ESK_FIYATI    = LS_ESK_SONUC-TTR_VRM_ESK_FIYATI  .
    LS_TOP_SATICI-HEDEF_FIYATI      = LS_ESK_SONUC-TTR_HEDEF_FIYATI    .
    LS_TOP_SATICI-ANLASMA_BRM_FYT   = LS_ESK_SONUC-TTR_ANLASMA_BRM_FYT .

    COLLECT LS_TOP_SATICI INTO LT_TOP_SATICI.
    ADD LS_TOP_SATICI-ANLASMA_BRM_FYT TO LV_TOPLAM_TUTAR.

  ENDLOOP.

  LOOP AT LT_TOP_SATICI INTO LS_TOP_SATICI.
    CLEAR:LS_SONUC_ORAN,
          LS_TOP_TUTAR,
          LV_MENSE.

    LS_SONUC_ORAN-DONEM          = LS_TOP_SATICI-DONEM.
    LS_SONUC_ORAN-LIFNR          = LS_TOP_SATICI-LIFNR.
    LS_SONUC_ORAN-BUYER          = LS_TOP_SATICI-BUYER.
    LS_SONUC_ORAN-MENSE          = LS_TOP_SATICI-MENSE.
    LS_SONUC_ORAN-KATEGORI       = LS_TOP_SATICI-KATEGORI.
    LS_SONUC_ORAN-UYGULANAN      = ( LS_TOP_SATICI-UYGULANAN_FIYAT  / LS_TOP_SATICI-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_SONUC_ORAN-ESKALASYON     = ( LS_TOP_SATICI-ESKALASYON_FIYAT / LS_TOP_SATICI-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_SONUC_ORAN-VRM_ESKALASYON = ( LS_TOP_SATICI-VRM_ESK_FIYATI   / LS_TOP_SATICI-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_SONUC_ORAN-HEDEF          = ( LS_TOP_SATICI-HEDEF_FIYATI     / LS_TOP_SATICI-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_SONUC_ORAN-ANLASMA        = ( LS_TOP_SATICI-ANLASMA_BRM_FYT  / LS_TOP_SATICI-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_SONUC_ORAN-PERFORMANS     = LS_SONUC_ORAN-HEDEF - LS_SONUC_ORAN-ANLASMA.
    LS_SONUC_ORAN-ESK_PERFORMANS = LS_SONUC_ORAN-UYGULANAN - LS_SONUC_ORAN-ANLASMA.
    LS_SONUC_ORAN-CIRO           = LS_TOP_SATICI-ANLASMA_BRM_FYT.
    LS_SONUC_ORAN-CIRO_PAYI      = ( LS_SONUC_ORAN-CIRO / LV_TOPLAM_TUTAR ) * 100.

    CLEAR:LS_LIFNR.
    READ TABLE LT_LIFNR INTO LS_LIFNR WITH TABLE KEY LIFNR = LS_SONUC_ORAN-LIFNR.
    IF SY-SUBRC EQ 0.
      LS_SONUC_ORAN-NAME1 = LS_LIFNR-NAME1.
    ENDIF.

    CLEAR: LS_BUYER.
    READ TABLE LT_BUYER INTO LS_BUYER WITH KEY BNAME = LS_SONUC_ORAN-BUYER.
    IF SY-SUBRC EQ 0.
      LS_SONUC_ORAN-BUYER_ADI = LS_BUYER-NAME_TEXT.
    ENDIF.

    APPEND LS_SONUC_ORAN TO LT_SONUC_ORAN. " gönderilecek

    IF I_MENSE EQ C_TUMU.
      LV_MENSE = C_TUMU.
    ELSE.
      LV_MENSE = LS_TOP_SATICI-MENSE.
    ENDIF.

    LS_TOP_TUTAR-DONEM            = LS_TOP_SATICI-DONEM.
    LS_TOP_TUTAR-MENSE            = LV_MENSE.
    LS_TOP_TUTAR-BAZ_BIRIM_FIYAT  = LS_TOP_SATICI-BAZ_BIRIM_FIYAT.
    LS_TOP_TUTAR-UYGULANAN_FIYAT  = LS_TOP_SATICI-UYGULANAN_FIYAT.
    LS_TOP_TUTAR-ESKALASYON_FIYAT = LS_TOP_SATICI-ESKALASYON_FIYAT.
    LS_TOP_TUTAR-VRM_ESK_FIYATI   = LS_TOP_SATICI-VRM_ESK_FIYATI.
    LS_TOP_TUTAR-HEDEF_FIYATI     = LS_TOP_SATICI-HEDEF_FIYATI.
    LS_TOP_TUTAR-ANLASMA_BRM_FYT  = LS_TOP_SATICI-ANLASMA_BRM_FYT.

    IF IT_KATEGORI[] IS NOT INITIAL.
      LS_TOP_TUTAR-KATEGORI       = LS_TOP_SATICI-KATEGORI.
    ENDIF.

    COLLECT LS_TOP_TUTAR INTO LT_TOP_TUTAR.

  ENDLOOP.

  IF I_BIRIKIMSIZ_KUMULATIF EQ 'X'.
    DATA: TABIX TYPE SY-TABIX.
    DATA: WA_TOP_TUTAR TYPE TY_TOP_TUTAR.

    LOOP AT LT_TOP_TUTAR INTO LS_TOP_TUTAR FROM 2.
      TABIX = SY-TABIX - 1.

      READ TABLE LT_TOP_TUTAR INTO WA_TOP_TUTAR INDEX TABIX.
      WA_TOP_TUTAR-DONEM = LS_TOP_TUTAR-DONEM.
      COLLECT WA_TOP_TUTAR INTO LT_TOP_TUTAR.
    ENDLOOP.
  ENDIF.

  LOOP AT LT_TOP_TUTAR INTO LS_TOP_TUTAR.
    CLEAR:LS_TOP_ORAN.

    LS_TOP_ORAN-DONEM          = LS_TOP_TUTAR-DONEM.
    LS_TOP_ORAN-MENSE          = LS_TOP_TUTAR-MENSE.
    LS_TOP_ORAN-KATEGORI       = LS_TOP_TUTAR-KATEGORI.
    LS_TOP_ORAN-UYGULANAN      = ( LS_TOP_TUTAR-UYGULANAN_FIYAT  / LS_TOP_TUTAR-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_TOP_ORAN-ESKALASYON     = ( LS_TOP_TUTAR-ESKALASYON_FIYAT / LS_TOP_TUTAR-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_TOP_ORAN-VRM_ESKALASYON = ( LS_TOP_TUTAR-VRM_ESK_FIYATI   / LS_TOP_TUTAR-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_TOP_ORAN-HEDEF          = ( LS_TOP_TUTAR-HEDEF_FIYATI     / LS_TOP_TUTAR-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_TOP_ORAN-ANLASMA        = ( LS_TOP_TUTAR-ANLASMA_BRM_FYT  / LS_TOP_TUTAR-BAZ_BIRIM_FIYAT - 1 ) * 100.
    LS_TOP_ORAN-PERFORMANS     = LS_TOP_ORAN-HEDEF - LS_TOP_ORAN-ANLASMA.
    LS_TOP_ORAN-ESK_PERFORMANS = LS_TOP_ORAN-UYGULANAN - LS_TOP_ORAN-ANLASMA.

    APPEND LS_TOP_ORAN TO LT_TOP_ORAN. " gönderilecek

  ENDLOOP.

  CLEAR:ET_SONUC[],
        ET_SATICI_SONUC[].

  DATA:LS_SONUC        TYPE ZMM_ESK_S_WEB_OZET,
       LS_SATICI_SONUC TYPE ZMM_ESK_S_WEB_SATICI.

  LOOP AT LT_TOP_ORAN INTO LS_TOP_ORAN.
    CLEAR:LS_SONUC.
    MOVE-CORRESPONDING LS_TOP_ORAN TO LS_SONUC.
    APPEND LS_SONUC TO ET_SONUC.
  ENDLOOP.

  LOOP AT LT_SONUC_ORAN INTO LS_SONUC_ORAN.
    CLEAR:LS_SATICI_SONUC.
    MOVE-CORRESPONDING LS_SONUC_ORAN TO LS_SATICI_SONUC.
    APPEND LS_SATICI_SONUC TO ET_SATICI_SONUC.
  ENDLOOP.

  SORT ET_SONUC BY DONEM.
  SORT ET_SATICI_SONUC BY DONEM LIFNR.

ENDFUNCTION.
